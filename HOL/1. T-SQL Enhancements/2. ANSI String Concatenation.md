DECLARE @UserId int

SELECT 'User ID ' + CONVERT(varchar(max), @UserId)
SELECT CONCAT('User ID ', @UserId)
SELECT 'User ID ' || @UserId		-- Like CONCAT, no conversion needed; like + a NULL resolves to NULL
GO
